extr short  random(short a, short b);
extr short len(string s);
extr bool writeOct(short n);


func short fact(short numb)
{
	var short result;
	result = 0b1;
	while (numb > 0b1)
	{
		result = result * numb;
		numb = numb - 0b1;
	}
	ret result;
}


proc task(string s1, string s2)
{
	var string dest;
	dest = s1 + s2;
	write dest;
	write $len(dest);
}

main
{
	var char ch;
	ch = 'a';
	write "Контрольный пример";

	write ch;
	write "Получим факториал от числа:";
	var short numb1;
	numb1 = 0o6;
	$writeOct(numb1);
	write "Результат:";
	write $fact(numb1);

	
	ch = 'b';
	write ch;
	write "Получим целое и остаток от деления 5 на 2:";
	var short a;
	var short b;
	var short c;
	a=0o5;
	b=0o2;
	write "Целое:";
	c = a/b;
	write c;
	write "Остаток:";
	c = a%b;
	write c;

	ch = 'c';
	write ch;
	c = $random(0b100,0b10) + 1b1;
	write c;

	ch='d';
	write ch;
	write "Сложим две строки и выведем их длину";
	var string s1;
	var string s2;
	s1 = "HELLO, ";
	s2 = "WORLD!";
	$task(s2,s1);

	var bool end;
	end = true;
	write "Контрольный пример окончен успешно:";
	write end;
}
